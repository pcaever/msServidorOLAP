package com.everis.msdemo.be.service;

import java.util.ArrayList;
import java.util.List;
import java.util.stream.Collectors;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.http.HttpEntity;
import org.springframework.http.HttpHeaders;
import org.springframework.http.HttpMethod;
import org.springframework.stereotype.Service;
import org.springframework.util.CollectionUtils;
import org.springframework.web.client.RestTemplate;

import com.everis.msdemo.be.dto.AdjuntoDTO;
import com.everis.msdemo.be.dto.DocumentosDTO;
import com.everis.msdemo.be.dto.GenericDTO;
import com.everis.msdemo.be.dto.PrincipalDTO;
import com.everis.msdemo.be.dto.UserDTO;
import com.everis.msdemo.be.exception.SiecaException;
import com.everis.msdemo.be.mapper.AdjuntosMapper;
import com.everis.msdemo.be.mapper.PrincipalMapper;
import com.everis.msdemo.be.msdto.PrincipalMSDTO;
import com.everis.msdemo.be.msdto.PrincipalMainMSDTO;
import com.everis.msdemo.be.msdto.PrincipalResponseMSDTO;

@Service
public class PrincipalService {
	@Value("${ws.backend.url}")
	private String url;
	
	@Value("${ws.backend.save.principal.path}")
	private String principalSavePath;
	
	public static final String HEADER_CONTENTTYPE = "Content-Type";
	public static final String HEADER_APPLICATIONJSON = "application/json; charset=UTF-8";
	
	private RestTemplate restTemplate = new RestTemplate();
	
	@Autowired
	private AdicionalService adicionalService;
    
	public PrincipalResponseMSDTO saveDemo(Integer usuarioCreacion, PrincipalMainMSDTO principalMainMSDTO) throws SiecaException {
		validateValuesPrincipal(principalMainMSDTO.getPrincipal());
		List<AdjuntoDTO> adjuntos = saveAdjunto(principalMainMSDTO, usuarioCreacion);
		savePrincipal(PrincipalMapper.INSTANCE.principalMSDTOToPrincipalDTO(principalMainMSDTO.getPrincipal(), adjuntos, usuarioCreacion));
		return PrincipalMapper.INSTANCE.principalMSDTOToPrincipalResponseMSDTO(principalMainMSDTO.getPrincipal());
	}
	
	public void validateValuesPrincipal(PrincipalMSDTO principalMSDTO) throws SiecaException {
    	List<GenericDTO> listaPaises = adicionalService.getCountries();
    	principalMSDTO.setPais(adicionalService.setGeneric(listaPaises, principalMSDTO.getPais(), "principalMSDTO.pais.code"));
    }
    
    public PrincipalDTO savePrincipal(PrincipalDTO principalDTO) {
    	HttpEntity<PrincipalDTO> entity = new HttpEntity<>(principalDTO, getHeaders());
		return restTemplate.exchange(String.format("%s%s", url, principalSavePath), HttpMethod.POST, entity, PrincipalDTO.class).getBody();
    }
    
    public List<AdjuntoDTO> saveAdjunto(PrincipalMainMSDTO principalMainMSDTO, Integer usuarioCreacion) throws SiecaException {
    	List<DocumentosDTO> adjuntosId = new ArrayList<>();
    	
    	if (!CollectionUtils.isEmpty(principalMainMSDTO.getDocumentosSoporte())) {
    		adjuntosId = adicionalService.saveAdjuntosBase64(principalMainMSDTO.getDocumentosSoporte().stream().map(mapAdjunto -> 
				AdjuntosMapper.INSTANCE.documentosMSDTOToDocumentosDTO(mapAdjunto, new UserDTO(usuarioCreacion, ""))).collect(Collectors.toList()));
    	}
    	
    	return adjuntosId.stream().map(mapAdjuntos -> AdjuntosMapper.INSTANCE.documentosDTOToAdjuntoDTO(mapAdjuntos, usuarioCreacion)).collect(Collectors.toList());
    }
    
    public HttpHeaders getHeaders() {
    	HttpHeaders headers = new HttpHeaders();
    	headers.set(HEADER_CONTENTTYPE, HEADER_APPLICATIONJSON);
		return headers;
    }
}
