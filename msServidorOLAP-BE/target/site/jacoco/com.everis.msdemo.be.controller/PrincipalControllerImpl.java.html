package com.everis.msdemo.be.controller;

import javax.validation.Valid;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpStatus;
import org.springframework.web.bind.annotation.CrossOrigin;
import org.springframework.web.bind.annotation.ResponseStatus;
import org.springframework.web.bind.annotation.RestController;

import com.everis.msdemo.be.constant.ConstantMsg;
import com.everis.msdemo.be.dto.MensajesErrorDTO;
import com.everis.msdemo.be.dto.ResponseDTO;
import com.everis.msdemo.be.exception.SiecaException;
import com.everis.msdemo.be.msdto.PrincipalMainMSDTO;
import com.everis.msdemo.be.service.AdicionalService;
import com.everis.msdemo.be.service.PrincipalService;

@RestController
@CrossOrigin(origins = "*")
public class PrincipalControllerImpl extends HandleErrorControllerImpl implements PrincipalController {
	@Autowired
	private PrincipalService principalService;
	
	@Autowired
	private AdicionalService adicionalService;

	@ResponseStatus(HttpStatus.CREATED)
	public ResponseDTO save(Integer usuarioCreacion, @Valid PrincipalMainMSDTO encabezadoMSDTO) throws SiecaException {
		ResponseDTO responseDTO = new ResponseDTO();
		responseDTO.setResponse(principalService.saveDemo(usuarioCreacion, encabezadoMSDTO));
		MensajesErrorDTO mensajeError = adicionalService.getMensajeError(ConstantMsg.CODE_201);
		responseDTO.getError().setCodError(mensajeError.getCodigoError());
		responseDTO.getError().setMessError(mensajeError.getMensajeError());
		return responseDTO;
	}
}
